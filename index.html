</head>

<body>

  <h2>Columbia Ferry Friday Departures — Summer 2025</h2>

  <div id="chart" style="width:100%;height:600px;"></div>


  <script>

    fetch("columbia_true_departures_sustained_2025.csv")

      .then(r => r.text())

      .then(text => {

        const rows = text.trim().split("\n").slice(1);

        const dates = [];

        const times = [];


        rows.forEach(r => {

          const cols = r.split(",");

          const dep = cols[3]; // true_departure_time_pt

          if (!dep) return;


          const d = new Date(dep);

          dates.push(cols[0]);

          times.push(d);

        });


        const trace = {

          x: dates,

          y: times,

          mode: "markers",

          marker: { size: 10 },

          text: times.map(t => t.toLocaleString()),

          hovertemplate: "%{text}<extra></extra>"

        };


        const layout = {

          yaxis: {

            title: "Departure Time (PT)",

            tickformat: "%H:%M"

          },

          xaxis: { title: "Friday" },

          title: "Columbia Ferry Departure Times"

        };


        Plotly.newPlot("chart", [trace], layout);

      });

  </script>

</body>
